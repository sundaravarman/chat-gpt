import pandas as pd

# Sample data
data = {
    'a': ['2023-01-15', '2023-03-22', '2022-12-31', '2023-06-10', '2024-07-15'],
    'b': [1, 2, 3, 4, 5],
    'c': [10, 20, 30, 40, 50],
    'd': [100, 200, 300, 400, 500],
    'e': [1000, 2000, 3000, 4000, 5000]
}
df = pd.DataFrame(data)

# Ensure the dates are in datetime format
df['a'] = pd.to_datetime(df['a'])

# Get the current date
current_date = pd.Timestamp.now().normalize()

# Filter the DataFrame to keep only the dates that haven't crossed the current date
filtered_df = df[df['a'] >= current_date]

print(filtered_df)
